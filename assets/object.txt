     //create object Helen
    let Helen = {
        //define object properties
        Firtsname:"Helen",
        Lastname: "Bell",
        age: 25,
        gender: "F",
        from: "Portugal",
        //is possible to define a function in property
        fullname: function(){
            return "Full name: " + this.Firtsname + " " + this.Lastname;
        }
    }

    //create object Adam
    let Adam = new Object();
    Adam.Firtsname = "Adam";
    Adam.Lastname = "Erno";
    Adam.birthday = "12-07-1994";
    Adam.fullname = () => {
        return Adam.Firtsname + " " + Adam.Lastname;
    }

    console.log("Name: " + Helen.Firtsname + "\nfrom: " + Helen.from + "\ngender: " + Helen.gender);
    // this wont print helen's full name, instead will return function definition
    console.log(Helen.fullname);

    console.log("Adam's birthday " + Adam.birthday);
